workspace "Nexa Container View" "Diagrama de Contenedores para el sistema Nexa" {

    model {
        # Personas (Mismos actores del contexto)
        staff = person "Commercial Coordination Staff" "Receive, interpret, and channel orders into the operational flow, reducing manual workload and ambiguity."
        b2bCustomer = person "B2B Customer" "Browses catalog, places orders, and tracks order status."
        distributorStaff = person "Distributor Staff" "Manages inventory, updates product catalog, and processes orders."
        visitorCustomer = person "Visitor – Customer" "External user exploring offered services and requesting cheese products."
        visitorDistributor = person "Visitor – Distributor Company" "External company user exploring services to distribute their cheeses."

        nexa = softwareSystem "Nexa" "Web platform for catalog and order management." {
            # Contenedores Internos
            webApp = container "Web App" "Provides all business functionality to internal staff and B2B customers." "Blazor WebAssembly"
            webSite = container "Web Site" "Public-facing landing page for visitors to explore Nexa services." "HTML, CSS, JavaScript"
            backendApi = container "Backend API" "Handles business logic and provides data via JSON/HTTPS." "ASP.NET Core Web API"
            database = container "Database" "Stores user, product, and order information." "MySQL" "Database"
        }

        # Sistemas Externos
        stripe = softwareSystem "Stripe" "Payment system" "External"
        googleCalendar = softwareSystem "Google Calendar" "Synchronizes delivery dates and logistics events with customers’ or internal staff calendars." "External"
        googleOAuth = softwareSystem "Google OAuth" "Manages user authentication and secure access to the system." "External"
        googleCloudStorage = softwareSystem "Google Cloud Storage" "Stores documents, product images, and order records securely in the cloud." "External"
        googleNotification = softwareSystem "Google Notification" "Sends email, notifications, and SMS." "External"

        # Relaciones: Personas -> Contenedores
        staff -> webApp "request"
        b2bCustomer -> webApp "request"
        distributorStaff -> webApp "request"
        visitorCustomer -> webSite "request"
        visitorDistributor -> webSite "request"

        # Relaciones entre Contenedores
        webSite -> webApp "request"
        webApp -> backendApi "request"
        backendApi -> database "Stores"

        # Relaciones: Backend -> Sistemas Externos
        backendApi -> stripe "pay"
        backendApi -> googleCalendar "uses"
        backendApi -> googleOAuth "Authenticate/receive"
        backendApi -> googleCloudStorage "uses"
        backendApi -> googleNotification "send"
    }

    views {
        container nexa "Containers" {
            include *
            autoLayout lr
        }

        styles {
            element "Container" {
                background #438dd5
                color #ffffff
            }
            element "Database" {
                shape Cylinder
            }
            element "External" {
                background #999999
                color #ffffff
            }
            element "Person" {
                shape Person
                background #08427b
                color #ffffff
            }
        }
    }
}
